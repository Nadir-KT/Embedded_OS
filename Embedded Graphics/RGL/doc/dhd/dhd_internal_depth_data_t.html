Summary

Internal type for integer access to depth data.

Syntax
typedef union tagdhd_internal_depth_data {
  struct dhd_depth_float32_t {
    dhd_float32_t m_z0;
    dhd_float32_t m_dzx, m_dzy;
    dhd_float32_t m_rhw0;
    dhd_float32_t m_dwx, m_dwy;
  } m_float32;
  struct dhd_depth_uint32_t {
    dhd_uint32_t m_z0_i;
    dhd_uint32_t m_dzx_i, m_dzy_i;
    dhd_uint32_t m_rhw0_i;
    dhd_uint32_t m_dwx_i, m_dwy_i;
  } m_uint32;
} dhd_internal_depth_data_t;
