dhd_buffer_t
Structs, Records, Enums | Send Feedback
Summary

A rectangular region of video memory.

Syntax
typedef struct tagdhd_buffer {
  dhd_vidmem_t m_memory;
  dhd_buffer_internal_t * m_internal;
  dhd_uint16_t m_width, m_height;
  dhd_uint16_t m_pitch;
  dhd_uint16_t m_bpp;
  dhd_uint16_t m_reserved;
  dhd_memory_usage_t m_usage;
  dhd_enum_t m_format;
  dhd_address_mode_t m_access;
  dhd_wrap_mode_t m_uaxis_wrap;
  dhd_wrap_mode_t m_vaxis_wrap;
} dhd_buffer_t;
