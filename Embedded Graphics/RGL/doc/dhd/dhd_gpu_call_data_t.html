D/AVE HD Driver - Internal Documentation
dhd_gpu_call_data_t
Structs, Records, Enums | Send Feedback
Summary

Generic structure for parameter passing.

Syntax
typedef union tagdhd_gpu_call_data {
  dhd_uint32_t m_response;
  struct tagget_instance_count {
    dhd_uint32_t * m_out;
  } m_get_instance_count;
  struct tagquery_instance {
    dhd_uint32_t m_instance;
    dhd_device_description_t * m_out;
  } m_query_instance;
  struct taginit {
    dhd_uint32_t m_instance;
    const dhd_gpu_config_t * m_config;
    dhd_uint8_t m_version_major_out;
    dhd_uint8_t m_version_minor_out;
    dhd_uint8_t m_revision_out;
  } m_init;
  struct tagshutdown {
    dhd_uint32_t m_instance;
  } m_shutdown;
  struct tagconnect {
    dhd_uint32_t m_instance;
    dhd_handle_t * m_out;
  } m_connect;
  struct tagdisconnect {
    dhd_handle_t m_client;
  } m_disconnect;
  struct tagalloc_mem {
    dhd_handle_t m_client;
    dhd_uint32_t m_size;
    dhd_uint32_t m_alignment;
    dhd_uint32_t m_tag;
    dhd_memory_usage_t m_usage;
    dhd_vidmem_t * m_result;
  } m_alloc_mem;
  struct tagfree_mem {
    dhd_handle_t m_client;
    dhd_vidmem_t * m_memory;
  } m_free_mem;
  struct tagcan_map_mem {
    dhd_handle_t m_client;
    dhd_vidmem_t * m_memory;
    dhd_enum_t m_access;
  } m_can_map_mem;
  struct tagmap_mem {
    dhd_handle_t m_client;
    dhd_vidmem_t * m_memory;
    dhd_enum_t m_access;
  } m_map_mem;
  struct tagunmap_mem {
    dhd_handle_t m_client;
    dhd_vidmem_t * m_memory;
    dhd_enum_t m_access;
  } m_unmap_mem;
  struct tagupload {
    dhd_handle_t m_client;
    const dhd_uint8_t * m_src;
    dhd_gpu_ptr_t m_dest;
    dhd_uint32_t m_size;
    dhd_bool_t m_tlistonly;
  } m_upload;
  struct tagdownload {
    dhd_handle_t m_client;
    dhd_gpu_ptr_t m_src;
    dhd_uint8_t * m_dest;
    dhd_uint32_t m_size;
  } m_download;
  struct tagcommit {
    dhd_handle_t m_client;
    struct tagdhd_gpu_job * m_job;
    struct tagdhd_gpu_job ** m_reclaimed_jobs;
  } m_commit;
  struct tagget_sync {
    dhd_handle_t m_client;
    dhd_uint32_t * m_jobid;
    dhd_uint32_t * m_clid;
  } m_get_sync;
  struct tagwait_for_job {
    dhd_uint32_t m_unused;
    dhd_handle_t m_client;
    dhd_jobid_t m_id;
  } m_wait_for_job;
  struct tagget_reclaimed_jobs {
    dhd_handle_t m_client;
    struct tagdhd_gpu_job ** m_out;
  } m_get_reclaimed_jobs;
  struct tagforce_yield {
    dhd_handle_t m_client;
  } m_force_yield;
  struct tagdownload_state {
    dhd_handle_t m_client;
    dhd_uint32_t * m_destination;
    dhd_uint32_t m_count;
    dhd_gpu_state_flags_t m_flags;
  } m_download_state;
  struct tagtlist_enable {
    dhd_handle_t m_client;
    const dhd_char_t * m_file;
  } m_tlist_enable;
  struct tagtlist_write {
    dhd_handle_t m_client;
    const dhd_char_t * m_string;
    dhd_bool_t m_to_kernel_tlist;
  } m_tlist_write;
  struct tagalloc_pc {
    dhd_handle_t m_client;
    dhd_uint8_t * m_pcid;
  } m_alloc_pc;
  struct tagfree_pc {
    dhd_handle_t m_client;
    dhd_uint8_t m_pcid;
  } m_free_pc;
  struct tagalloc_clut {
    dhd_handle_t m_client;
    dhd_uint32_t m_size;
    dhd_uint32_t * m_clutindex;
  } m_alloc_clut;
  struct tagfree_clut {
    dhd_handle_t m_client;
    dhd_uint32_t m_size;
    dhd_uint32_t m_clutindex;
  } m_free_clut;
  struct tagresume {
    dhd_uint32_t m_instance;
  } m_resume;
  struct tagcan_access_mem {
    dhd_handle_t m_client;
    dhd_uint32_t m_size;
    dhd_uint8_t * m_cpu_address;
  } m_can_access_mem;
} dhd_gpu_call_data_t;
Description

All kernel functions have the same signature: dhd_uint32_t function(dhd_gpu_call_data_t *a_parameters)
